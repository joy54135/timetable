Version 13.x (Next Gen)"); } // --- 保存・読込処理 --- function saveState(stateObj) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName("App_State"); if (!sheet) { sheet = ss.insertSheet("App_State"); } sheet.showSheet(); sheet.getRange("A1").setValue(JSON.stringify(stateObj)); sheet.getRange("B1").setValue("最終更新: " + Utilities.formatDate(new Date(), "JST", "yyyy/MM/dd HH:mm:ss")); return { success: true }; } catch (e) { return { success: false, message: "保存失敗: " + e.message }; } } function loadState() { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheet = ss.getSheetByName("App_State"); if (!sheet) return { success: false, message: "保存されたデータがありません" }; const jsonStr = sheet.getRange("A1").getValue(); if (!jsonStr) return { success: false, message: "データが空です" }; return { success: true, state: JSON.parse(jsonStr) }; } catch (e) { return { success: false, message: "読込失敗: " + e.message }; } } // --- エクスポート処理 --- function exportToSpreadsheet(schedule, data) { try { const ss = SpreadsheetApp.getActiveSpreadsheet(); const sheetName = "確定時間割_" + Utilities.formatDate(new Date(), "JST", "MMdd_HHmm"); const sheet = ss.insertSheet(sheetName); const header = ["クラス"]; ['月', '火', '水', '木', '金'].forEach(d => { [1, 2, 3, 4, 5, 6].forEach(p => header.push(`${d}${p}`)); }); sheet.getRange(1, 1, 1, header.length).setValues([header]).setFontWeight("bold").setBackground("#f0f2f9"); let row = 2; Object.entries(data.grades).forEach(([g, n]) => { for (let c = 1; c <= n; c++) { const className = `${g}-${c}`; const rowData = [className]; ['月', '火', '水', '木', '金'].forEach(d => { [1, 2, 3, 4, 5, 6].forEach(p => { let cellText = ""; if (schedule[d] && schedule[d][p]) { const lesson = schedule[d][p].find(l => l.targets.includes(className)); if (lesson) { cellText = `${lesson.subject}\n(${lesson.teacherName})`; if (lesson.room !== '通常教室') cellText += `\n@${lesson.room}`; } } rowData.push(cellText); }); }); sheet.getRange(row, 1, 1, rowData.length).setValues([rowData]); row++; } }); sheet.setRowHeights(2, row - 2, 50); sheet.setColumnWidths(2, header.length - 1, 80); sheet.getRange(1, 1, row - 1, header.length).setBorder(true, true, true, true, true, true).setHorizontalAlignment("center").setVerticalAlignment("middle"); return { success: true }; } catch (error) { return { success: false, message: error.message }; } } // --- 自動生成アルゴリズム --- function generateSchedule(state) { const startTime = Date.now(); const TIME_LIMIT = 240000; // 4分で強制終了回避 const DAYS = ['月', '火', '水', '木', '金']; const PERIODS = [1, 2, 3, 4, 5, 6]; const schedule = {}; DAYS.forEach(d => { schedule[d] = {}; PERIODS.forEach(p => schedule[d][p] = []); }); const teacherMap = {}; if (state.teachers) { state.teachers.forEach(t => teacherMap[t.id] = t.name); } // 1. 全コマの抽出 let lessons = []; // (A) 特殊ブロック if (state.specialBlocks) { state.specialBlocks.forEach(b => { for (let i = 0; i < b.hours; i++) { lessons.push({ id: `sp_${b.id}_${i}`, subjectId: b.subjectId, subject: state.subjects.find(s=>s.id===b.subjectId)?.name || '特殊', targets: b.targets || [], teacherIds: (b.teachers || []).map(t => t.id), teacherName: (b.teachers || []).map(t => teacherMap[t.id]).join(', ') || '未定', room: b.room || '通常教室', isSp: true }); } }); } // (B) クラス別担当（通常授業） if (state.classAssignments) { Object.keys(state.classAssignments).forEach(cls => { const grade = cls.split('-')[0]; const data = state.classAssignments[cls]; state.subjects.forEach(sub => { const hours = parseInt(sub.stdHours[grade]) || 0; if (hours === 0) return; // 特殊ブロックで設定されている教科はスキップ if (state.specialBlocks && state.specialBlocks.some(b => b.subjectId === sub.id && b.targets.includes(cls))) return; const tId = data.overrides && data.overrides[sub.id] && data.overrides[sub.id].teacherId ? data.overrides[sub.id].teacherId : data.homeroom; if (!tId) return; // 担任未定など const roomName = data.overrides && data.overrides[sub.id] && data.overrides[sub.id].room ? data.overrides[sub.id].room : (sub.defaultRoom || '通常教室'); for (let i = 0; i < hours; i++) { lessons.push({ id: `n_${cls}_${sub.id}_${i}`, subjectId: sub.id, subject: sub.name, targets: [cls], teacherIds: [tId], teacherName: teacherMap[tId] || '未定', room: roomName, isSp: false }); } }); }); } // 2. 優先度ソート（条件が厳しい順: 合同・特別教室・複数教員を優先） lessons.sort((a, b) => { let scoreA = (a.targets.length * 10) + (a.room !== '通常教室' ? 5 : 0) + a.teacherIds.length; let scoreB = (b.targets.length * 10) + (b.room !== '通常教室' ? 5 : 0) + b.teacherIds.length; return scoreB - scoreA; }); const unassigned = []; // 3. 配置実行 for (let i = 0; i < lessons.length; i++) { if (Date.now() - startTime > TIME_LIMIT) { unassigned.push(lessons[i]); continue; } const lesson = lessons[i]; let placed = false; let candidates = []; DAYS.forEach(d => PERIODS.forEach(p => candidates.push({d, p}))); candidates = candidates.sort(() => Math.random() - 0.5); // 【高度ルール】最終コマ固定 const isLastPeriodSub = state.advancedRules && state.advancedRules.lastPeriodSubjects && state.advancedRules.lastPeriodSubjects.includes(lesson.subject); if (isLastPeriodSub) { candidates = candidates.filter(c => { let isLast = true; lesson.targets.forEach(tgt => { const g = tgt.split('-')[0]; if (state.periods && state.periods[g] && state.periods[g][c.d] !== c.p) isLast = false; }); return isLast; }); } for (let c = 0; c < candidates.length; c++) { const { d, p } = candidates[c]; // ① 学年の時数枠内か let outOfBounds = false; for (let target of lesson.targets) { const grade = target.split('-')[0]; if (state.periods && state.periods[grade] && p > (state.periods[grade][d] || 0)) { outOfBounds = true; break; } } if (outOfBounds) continue; // ② 全校ブロック if (state.globalBlocks && state.globalBlocks[`${d}-${p}`]) continue; // ③ 教員衝突＆不可枠 let teacherConflict = false; for (let tId of lesson.teacherIds) { const t = state.teachers.find(x => x.id === tId); if (t && t.ngTimes && t.ngTimes[`${d}-${p}`]) { teacherConflict = true; break; } const existing = schedule[d][p].some(l => l.teacherIds.includes(tId)); if (existing) { teacherConflict = true; break; } } if (teacherConflict) continue; // ④ クラス衝突 let classConflict = false; for (let target of lesson.targets) { const existing = schedule[d][p].some(l => l.targets.includes(target)); if (existing) { classConflict = true; break; } } if (classConflict) continue; // ⑤ 教室キャパ if (lesson.room !== '通常教室') { const r = state.rooms.find(x => x.name === lesson.room); if (r) { if (r.ngTimes && r.ngTimes[`${d}-${p}`]) continue; const usedCount = schedule[d][p].filter(l => l.room === lesson.room).length; if (usedCount >= (parseInt(r.capacity) || 1)) continue; } } // ⑥ 1日1科目 if (state.advancedRules && state.advancedRules.limitOneSubjectPerDay) { let subConflict = false; for (let per = 1; per <= 6; per++) { if (schedule[d][per].some(l => l.subjectId === lesson.subjectId && l.targets.some(t => lesson.targets.includes(t)))) { subConflict = true; break; } } if (subConflict) continue; } schedule[d][p].push(lesson); placed = true; break; } if (!placed) { unassigned.push(lesson); } } return { schedule: schedule, unassigned: unassigned }; }
